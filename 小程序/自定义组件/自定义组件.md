#### 组件 Page

```java
// component.json
{
  "component": true,
  "usingComponents": {}
}

// component.js
// properties 代表的是在自定义组件中的attr
// src是要动态传入的:
// <component src="{{...}}"></component>
properties: {
  src: {
    type: String,
    value: ''
  }
},
   
// component.wxss	同其他的wxss一样
// component.wxml 	
<view>
  <image src='{{src}}'></image>
</view>

// 需要使用到的地方
// index.json
{
  "usingComponents": {
    "header-component": "../../components/header-component/header-component"
  }
}
// index.wxml
<header-component src="{{src}}">
</header-component>
```

#### 增加函数

```java
// 在component.js中加入
methods: {
  showHeaderInfo() {
    console.log(this.properties.src);
    console.log(this.data.src);
    console.log("this.src: ", this.src);
  }
}

// 在引用的index.wxml中
<header-component 
  src="{{src}}" 
  id="header-component"
  bindtap="showHeaderInfo"
  >
</header-component>

// 在引用的index.js中
onReady: function () {
  this.headerComponent = this.selectComponent("#header-component");
},
// 在需要的地方调用
this.headerComponent.showHeaderInfo();

```

##### Slot(插槽)

> 插槽的使用方式

```java
/// component.wxml
<header-component 
  src="{{src}}" 
  id="header-component"
  bindtap="showHeaderInfo"
  >
  <view slot="before">solt before</view>
  <view slot="after">solt after</view>
</header-component>

// 在引用的index.wxml中
<view>
  <slot name="before"></slot>
  <image src='{{src}}'></image>
  <slot name="after"></slot>
</view>
```
